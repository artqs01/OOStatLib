139template<Interval DataType, Ratio CalcType>
140CalcType variance<DataType, CalcType>::evaluate(const single_container<DataType>& data)
141{
142 CalcType avreage = mean<DataType, CalcType>::evaluate(data);
143 CalcType sum = 0;
144 for (const auto& value : data) {
145 sum += (value - avreage) * (value - avreage);
146 }
147 return sum / (data.size() - 1);
148}

279template<Interval DataType, Ratio CalcType>
280bool shapiro_wilk<DataType, CalcType>::evaluate(single_container<DataType>& data, significance alfa)
281{
282 CalcType w = 0.0;
283 std::sort(data.begin(), data.end());
284 for (size_t i = 0; i < data.size() / 2; i++) {
285 w += (data[data.size() - 1 - i] - data[i]) *
286 table_shapiro_wilk<CalcType>::get_coefficient(data.size(), i);
287 }
288 w *= w;
289 w /= (variance<DataType, CalcType>::evaluate(data) * (data.size() - 1));
290 return w > table_shapiro_wilk<CalcType>::get_p_value(data.size(), alfa);
291}

375template<Interval DataType, Ratio CalcType>
376bool mean_significance_test<DataType, CalcType>::evaluate(
377 single_container<DataType>& data,
378 CalcType _mean,
379 significance alfa)
380{
381 bool if_normal = shapiro_wilk<DataType, CalcType>::evaluate(data, alfa);
382 assert(if_normal);
383 CalcType t = (mean<DataType, CalcType>::evaluate(data) - _mean) * std::sqrt(data.size()) /
