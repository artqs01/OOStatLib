87bool pair_observations_test<DataType1, DataType2, CalcType>::evaluate(paired_container<DataType1, DataType2> &data, significance alfa)
88{
89 using DifferenceType = std::conditional_t<std::is_floating_point_v<DataType1> && std::is_floating_point_v<DataType2>,
90 std::conditional_t<sizeof(DataType1) >= sizeof(DataType2),
91 DataType1,
92 DataType2
93 >,
94 std::conditional_t<std::is_floating_point_v<DataType1>,
95 DataType1,
96 DataType2
97 >
98 >;
99 std::vector<DifferenceType> difference(data.size());
100 for (const auto& pair : data) {
101 difference.push_back(pair.second - pair.first);
102 }
103 CalcType t = mean<DifferenceType, CalcType>::evaluate(difference) * std::sqrt(difference.size()) /
104 standard_deviation<DifferenceType, CalcType>::evaluate(difference);
105 return pdf<CalcType>::t(std::abs(t), data.size() - 1) < significance_logic<CalcType>::to_floating_point(alfa);
106}
